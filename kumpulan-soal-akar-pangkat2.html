q² - 8q - 8 = 0
q1,2 = (8 + √ (64 - 128)) / 4

     = (8 + 8√-1) /4

[tex]$\begin{align} (x_1)^2+(x_2)^2&=(x_1+x_2)^2-2x_1x_2\\4&=\left(-\frac{1}{2}q\right)^2-2\times\frac{q-1}{2}\\4&=\frac{1}{4}q^2-q+1\\16&=q^2-4q+4\\16&=(q-2)^2\\\pm4&=q-2\\\\4&=q_1-2\\q_1&=6\\\\-4&=q_2-2\\q_2&=-2 \end{align}[/tex]
